###  03 参考基因组准备

```bash

################################################################
## 参考基因组准备
## 功能：下载参考基因组相关文件
## 参数说明：
##   axel参数:
##     -n 10 : 使用10个连接加速下载
##     -o 下载后的文件名
##   gunzip : 解压基因组文件
## 文件说明:
##   dna.primary_assembly.fa : 主要基因组序列
##   dna.toplevel.fa.gz ： 完整组装的基因组序列
##   cdna.all.fa : 全部转录本序列
##   .gtf.gz : 基因结构注释文件
################################################################
## 参考基因组准备:注意参考基因组版本信息
# 动物下载，Ensembl：http://asia.ensembl.org/index.html
# 植物下载，https://plants.ensembl.org/index.html
# 文章给出的地址下载

###拟南芥
# 进入到参考基因组目录
cd $workdir/1database/TAIR10
# cd /home/data/t150618/workdir/1database/TAIR10
# 下载基因组序列axel  curl  
nohup axel -n 10 -o dna.fa.gz https://ftp.ensemblgenomes.ebi.ac.uk/pub/plants/release-61/fasta/arabidopsis_thaliana/dna/Arabidopsis_thaliana.TAIR10.dna.toplevel.fa.gz >dna.log &
# 下载转录组序列
nohup axel -n 10 -o cdna.fa.gz https://ftp.ensemblgenomes.ebi.ac.uk/pub/plants/release-61/fasta/arabidopsis_thaliana/cdna/Arabidopsis_thaliana.TAIR10.cdna.all.fa.gz >rna.log &

# 下载蛋白质序列文件
nohup axel -n 10 -o pep.fa.gz https://ftp.ensemblgenomes.ebi.ac.uk/pub/plants/release-61/fasta/arabidopsis_thaliana/pep/Arabidopsis_thaliana.TAIR10.pep.all.fa.gz >pep.log &

# 下载基因组注释文件
nohup axel -n 10 -o dna.gtf.gz https://ftp.ensemblgenomes.ebi.ac.uk/pub/plants/release-61/gtf/arabidopsis_thaliana/Arabidopsis_thaliana.TAIR10.61.gtf.gz >gtf.log &
nohup axel -n 10 -o dna.gff3.gz https://ftp.ensemblgenomes.ebi.ac.uk/pub/plants/release-61/gff3/arabidopsis_thaliana/Arabidopsis_thaliana.TAIR10.61.gff3.gz >gff.log&






### 人类
# 进入到参考基因组目录
cd $workdir/1database/GRCh38.111

# 下载基因组序列axel  curl  
nohup axel -n 10 -o dna.fa.gz https://ftp.ensembl.org/pub/release-111/fasta/homo_sapiens/dna/Homo_sapiens.GRCh38.dna.primary_assembly.fa.gz >dna.log &

# 下载转录组序列
nohup axel -n 10 -o cdna.fa.gz https://ftp.ensembl.org/pub/release-111/fasta/homo_sapiens/cdna/Homo_sapiens.GRCh38.cdna.all.fa.gz >rna.log &


# 下载蛋白质序列文件
nohup axel -n 10 -o pep.fa.gz https://ftp.ensembl.org/pub/release-111/fasta/homo_sapiens/pep/Homo_sapiens.GRCh38.pep.all.fa.gz >pep.log &


# 下载基因组注释文件
nohup axel -n 10 -o dna.gtf.gz https://ftp.ensembl.org/pub/release-111/gtf/homo_sapiens/Homo_sapiens.GRCh38.111.gtf.gz >gtf.log &
nohup axel -n 10 -o dna.gff3.gz https://ftp.ensembl.org/pub/release-111/gff3/homo_sapiens/Homo_sapiens.GRCh38.111.gff3.gz >gff.log&
```
